// project demo rtl test,
// functions L2 switching, table_rank_calc_mode, strict priority, and wrr rank calc,




// ====================================================
// table rule for Switching mode to apply l1 switching(for wire debug), or l2 swithing, 
// table name: table_switch_mode
// key       : ethernet.srcAddr
// action    : set_switch_mode
// action param: 2bit vector switching mode, 0 for l1 switching, 1 for l2 switching


// the following 4 rule is used to forwarding for sfp port

//table_cam_add_entry table_switch_mode set_switch_mode 0x011111111101 => 1

//table_cam_add_entry table_switch_mode set_switch_mode 0xab00000000ab => 0

table_cam_add_entry table_switch_mode set_switch_mode 0x013333333301 => 1
table_cam_add_entry table_switch_mode set_switch_mode 0x014444444401 => 1


// ====================================================
// table rule for L1 swithing
// table name: table_l1_switch
// key       : sume_metadata.src_port
// action    : l2_forward
// action param: 8bit vector for output port


// the following 4 rule is used to forwarding for sfp port
//table_cam_add_entry table_l1_switch l2_forward 0b00000001 => 0b00000100
//table_cam_add_entry table_l1_switch l2_forward 0b00000100 => 0b00010000
//table_cam_add_entry table_l1_switch l2_forward 0b00010000 => 0b01000000
 
// ====================================================
// table rule for l2 switching, 
// table name: table_l2_switch
// key       : ethernet.dstAddr
// action    : l2_forward
// action param: 8bit vector for output port


// the following 4 rule is used to forwarding for sfp port
//table_cam_add_entry table_l2_switch l2_forward 0x010000000001 => 0b00000001
//table_cam_add_entry table_l2_switch l2_forward 0x020000000002 => 0b00000100
table_cam_add_entry table_l2_switch l2_forward 0x030000000003 => 0b00010000
table_cam_add_entry table_l2_switch l2_forward 0x040000000004 => 0b00010000


// ====================================================
// table rule for rank calc look up, 
// table name: table_rank_calc_mode
// key       : sume.dst_port
// action    : set_calc_mode
// action param: bit<1> strict, bit<1> wrr, bit<1> drr, bit<1>, wfq

table_cam_add_entry table_rank_calc_mode set_calc_mode 0b00010000 => 0 1 0 0

// ====================================================
// table rule for strict priority, 
// table name: table_strict_priority
// key       : ethernet.srcAddr
// action    : set_strict_rank
// action param: bit<32> pifo_info

// rank for 1
table_cam_add_entry table_strict_priority set_strict_rank 0x011111111101 => 0x80001000 
// rank for 5
table_cam_add_entry table_strict_priority set_strict_rank 0x012222222201 => 0x80005000 



// ====================================================
// table rule for wrr class lookup 
// table name: table_wrr_class_lookup
// key       : ethernet.srcAddr
// action    : set_wrr_class
// action param: bit<5> class

table_cam_add_entry table_wrr_class_lookup set_wrr_class 0x013333333301 => 2
table_cam_add_entry table_wrr_class_lookup set_wrr_class 0x014444444401 => 1
